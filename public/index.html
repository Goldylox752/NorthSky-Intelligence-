     async function executeRip() {
    const url = document.getElementById('urlInput').value;
    const btn = document.getElementById('ripBtn');
    const preview = document.getElementById('resultPreview');
    const output = document.getElementById('jsonOutput');

    if (!url) return alert("Enter a URL");

    // UI Loading State
    btn.disabled = true;
    btn.innerHTML = `🌀 RIPPING...`;
    preview.classList.remove('hidden');
    output.innerText = "Queuing task in background...";

    try {
        // 1. Send to Queue
        const res = await fetch(`/rip?url=${encodeURIComponent(url)}`);
        const { jobId } = await res.json();

        // 2. Start Polling for the Result
        pollStatus(jobId);
    } catch (err) {
        output.innerText = "Failed to queue job.";
        btn.disabled = false;
        btn.innerText = "RIP";
    }
}
